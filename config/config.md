## 配置文件使用参数说明
---
配置文件config.ini 各项参数注释如下：  
配置项后带//* 注释说明的可能是需要你去定制修改的参数

#### 通用配置
[common]  
unit=M　　　　　　//磁盘内存显示单位,可选项有G/M/K  

#### 主机配置信息
[host]  
ip=127.0.0.1　　　//* 填写服务器具体IP地址  

#### 处理器配置
[cpu]  
onboot=yes　　　　//是否启用CPU监控,yes表示启用,no表示不启用监控  
percent=90　　　　//* CPU使用率阈值,此处表示CPU使用率为90%,当监控值大于这个值便会报警  
thread_per_cpu=2　//CPU每个逻辑核心运行的负载阈值,当监控值大于这个值便会报警  

#### 磁盘配置
[disk]  
onboot=yes　　　　//是否启用磁盘监控,yes表示启用,no表示不启用监控  
percent=80　　　　//* 磁盘分区使用率阈值,此处表示磁盘空间使用率为80%,当监控值大于这个值便会报警  
available_G=5　　　//磁盘分区可用大小阈值,单位默认为G,此处表示最小磁盘可用空间为5G,当监控值小于这个值便会报警  
main_dir=/　　　　//表示系统根分区/,如有其它分区需要监控要在下面添加  

#### 内存配置
[memory]  
onboot=yes　　　　//是否启用内存监控,yes表示启用,no表示不启用监控  
percent=80　　　　//* 内存使用率阈值,此处表示内存使用率为80%,当监控值大于这个值便会报警  

#### 磁盘读写I/O配置
[input-output]  
onboot=yes　　　　//是否启用磁盘IO监控,yes表示启用,no表示不启用监控  
device=sda　　　　//* 磁盘设备名  
await=15　　　　　//* IO操作的平均等待时间阈值,单位为ms,此处表示15ms,当监控值大于这个值便会报警  
util=85　　　　　　//* 设备使用率阈值,表示磁盘的读写繁忙程度,此处表示磁盘读写使用率为85%,当监控值大于这个值便会报警  

#### nginx配置
[nginx]  
onboot=no　　　　//是否启用nginx监控,yes表示启用,no表示不启用监控  
port=80　　　　　//nginx服务的监听端口
